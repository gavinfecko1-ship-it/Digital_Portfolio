Intelligent Grocery & Inventory Tracking App

Smart solution for college students to manage groceries and reduce food waste

April 2025 Python, JavaScript, Flask Solo Project

Project Overview

The Problem

College students struggle with food waste, budget management, and remembering what groceries they have. Studies show students waste an average of $1,500 worth of food annually due to poor inventory tracking.

My Solution

A web and mobile application that automatically tracks grocery inventory, sends expiration alerts, suggests recipes based on available ingredients, and helps with budget planning.

Target Users

College students, young professionals, and anyone looking to reduce food waste and manage grocery budgets more effectively.

Key Features

Expiration Alerts

Automated notifications when items are about to expire, with recipe suggestions to use them up.

Budget Tracking

Track grocery spending across multiple stores with visual analytics and spending patterns.

Recipe Suggestions

AI-powered recipe recommendations based on inventory and dietary preferences.

Store Comparison

Compare prices across local grocery stores to find the best deals.

Expiration Alert System

class AlertSystem:
    def __init__(self, db_connection):
        self.db = db_connection
    
    def check_expiring_items(self, days_threshold=3):
        """Find items expiring within specified days"""
        query = """
        SELECT u.email, i.name, i.expiration_date, DATEDIFF(i.expiration_date, CURDATE()) as days_left
        FROM items i
        JOIN users u ON i.user_id = u.id
        WHERE i.expiration_date IS NOT NULL 
        AND DATEDIFF(i.expiration_date, CURDATE()) BETWEEN 0 AND %s
        AND i.quantity > 0
        """
        
        with self.db.cursor() as cursor:
            cursor.execute(query, (days_threshold,))
            expiring_items = cursor.fetchall()
        
        # Group items by user
        user_alerts = {}
        for item in expiring_items:
            email = item['email']
            if email not in user_alerts:
                user_alerts[email] = []
            user_alerts[email].append(item)
        
        return user_alerts
    
    def generate_recipe_suggestions(self, items):
        """Generate recipe suggestions based on expiring items"""
        suggestions = []
        for item in items:
            # Query recipes that use this ingredient
            recipe_query = """
            SELECT id, name, prep_time, difficulty
            FROM recipes
            WHERE ingredients LIKE %s
            LIMIT 3
            """
            
            with self.db.cursor() as cursor:
                cursor.execute(recipe_query, (f'%{item["name"]}%',))
                recipes = cursor.fetchall()
                suggestions.extend(recipes)
        
        return suggestions

Budget Tracking Algorithm

class BudgetManager:
    def __init__(self, user_id, db_connection):
        self.user_id = user_id
        self.db = db_connection
    
    def analyze_spending(self, timeframe='monthly'):
        """Analyze spending patterns"""
        if timeframe == 'weekly':
            interval = 'INTERVAL 7 DAY'
        elif timeframe == 'monthly':
            interval = 'INTERVAL 30 DAY'
        else:
            interval = 'INTERVAL 1 YEAR'
        
        query = f"""
        SELECT 
            DATE(purchase_date) as date,
            category,
            SUM(price * quantity) as total_spent,
            COUNT(*) as item_count
        FROM items
        WHERE user_id = %s 
        AND purchase_date >= DATE_SUB(CURDATE(), {interval})
        GROUP BY DATE(purchase_date), category
        ORDER BY date DESC
        """
        
        with self.db.cursor() as cursor:
            cursor.execute(query, (self.user_id,))
            spending_data = cursor.fetchall()
        
        return self.calculate_trends(spending_data)
    
    def calculate_trends(self, data):
        """Calculate spending trends and predictions"""
        trends = {
            'total_spent': sum(item['total_spent'] for item in data),
            'average_daily': 0,
            'projected_monthly': 0,
            'top_categories': []
        }
        
        if data:
            daily_totals = {}
            category_totals = {}
            
            for item in data:
                # Aggregate by day
                day = item['date']
                daily_totals[day] = daily_totals.get(day, 0) + item['total_spent']
                
                # Aggregate by category
                category = item['category']
                category_totals[category] = category_totals.get(category, 0) + item['total_spent']
            
            # Calculate averages and projections
            num_days = len(daily_totals)
            trends['average_daily'] = trends['total_spent'] / max(num_days, 1)
            trends['projected_monthly'] = trends['average_daily'] * 30
            
            # Get top spending categories
            trends['top_categories'] = sorted(
                category_totals.items(),
                key=lambda x: x[1],
                reverse=True
            )[:5]
        
        return trends
