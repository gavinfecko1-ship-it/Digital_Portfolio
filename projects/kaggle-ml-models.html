1. Data Exploration & Cleaning

class DataPreprocessor:
    def __init__(self):
        self.imputers = {}
        self.scalers = {}
        self.encoders = {}
    
    def comprehensive_eda(self, df):
        """Perform exploratory data analysis"""
        eda_report = {
            'missing_values': df.isnull().sum(),
            'data_types': df.dtypes,
            'descriptive_stats': df.describe(),
            'correlation_matrix': df.corr(),
            'skewness': df.skew(),
            'outliers': self.detect_outliers(df)
        }
        return eda_report
    
    def handle_missing_values(self, df, strategy='advanced'):
        """Advanced missing value handling"""
        if strategy == 'advanced':
            # Use machine learning to predict missing values
            for col in df.columns[df.isnull().any()]:
                if df[col].dtype == 'object':
                    # Categorical imputation
                    df[col] = df[col].fillna(df[col].mode()[0])
                else:
                    # Numerical imputation with k-NN
                    from sklearn.impute import KNNImputer
                    imputer = KNNImputer(n_neighbors=5)
                    df[[col]] = imputer.fit_transform(df[[col]])
        return df
    
    def detect_outliers(self, df, method='iqr'):
        """Detect and handle outliers"""
        outliers = {}
        for col in df.select_dtypes(include=[np.number]).columns:
            if method == 'iqr':
                Q1 = df[col].quantile(0.25)
                Q3 = df[col].quantile(0.75)
                IQR = Q3 - Q1
                lower_bound = Q1 - 1.5 * IQR
                upper_bound = Q3 + 1.5 * IQR
                col_outliers = df[(df[col] < lower_bound) | (df[col] > upper_bound)]
                outliers[col] = len(col_outliers)
        return outliers

2. Feature Engineering

class FeatureEngineer:
    def __init__(self):
        self.feature_importance = {}
    
    def create_interaction_features(self, df):
        """Create interaction features between variables"""
        numerical_cols = df.select_dtypes(include=[np.number]).columns
        interaction_features = pd.DataFrame(index=df.index)
        
        # Create pairwise interactions
        for i, col1 in enumerate(numerical_cols):
            for col2 in numerical_cols[i+1:]:
                interaction_name = f"{col1}_x_{col2}"
                interaction_features[interaction_name] = df[col1] * df[col2]
        
        return pd.concat([df, interaction_features], axis=1)
    
    def create_polynomial_features(self, df, degree=2):
        """Create polynomial features"""
        from sklearn.preprocessing import PolynomialFeatures
        poly = PolynomialFeatures(degree=degree, include_bias=False)
        numerical_data = df.select_dtypes(include=[np.number])
        poly_features = poly.fit_transform(numerical_data)
        poly_df = pd.DataFrame(
            poly_features,
            columns=poly.get_feature_names_out(numerical_data.columns)
        )
        return pd.concat([df, poly_df], axis=1)
    
    def create_time_based_features(self, df, date_col):
        """Extract time-based features from date columns"""
        df['year'] = df[date_col].dt.year
        df['month'] = df[date_col].dt.month
        df['quarter'] = df[date_col].dt.quarter
        df['day_of_week'] = df[date_col].dt.dayofweek
        df['is_weekend'] = df['day_of_week'].isin([5, 6]).astype(int)
        df['day_of_year'] = df[date_col].dt.dayofyear
        return df

Ensemble Model Architecture

class EnsembleModel:
    def __init__(self):
        self.base_models = []
        self.meta_model = None
        self.cv_scores = {}
    
    def build_ensemble(self, X_train, y_train):
        """Build stacking ensemble model"""
        from sklearn.ensemble import StackingRegressor
        from sklearn.linear_model import Ridge
        from sklearn.ensemble import RandomForestRegressor, GradientBoostingRegressor
        from xgboost import XGBRegressor
        from lightgbm import LGBMRegressor
        
        # Define base models
        base_models = [
            ('rf', RandomForestRegressor(
                n_estimators=100,
                max_depth=10,
                random_state=42
            )),
            ('gb', GradientBoostingRegressor(
                n_estimators=100,
                learning_rate=0.1,
                random_state=42
            )),
            ('xgb', XGBRegressor(
                n_estimators=100,
                max_depth=6,
                learning_rate=0.1,
                random_state=42
            )),
            ('lgb', LGBMRegressor(
                n_estimators=100,
                learning_rate=0.1,
                random_state=42
            ))
        ]
        
        # Define meta-model
        meta_model = Ridge(alpha=1.0)
        
        # Create stacking ensemble
        self.ensemble = StackingRegressor(
            estimators=base_models,
            final_estimator=meta_model,
            cv=5,
            passthrough=True
        )
        
        return self.ensemble
    
    def hyperparameter_tuning(self, model, param_grid, X, y):
        """Advanced hyperparameter optimization"""
        from sklearn.model_selection import RandomizedSearchCV
        
        # Use Bayesian optimization for faster convergence
        from skopt import BayesSearchCV
        
        opt = BayesSearchCV(
            model,
            param_grid,
            n_iter=50,
            cv=5,
            scoring='neg_mean_squared_error',
            random_state=42,
            n_jobs=-1
        )
        
        opt.fit(X, y)
        return opt.best_estimator_, opt.best_score_
    
    def feature_selection(self, X, y, method='importance'):
        """Select most important features"""
        if method == 'importance':
            from sklearn.ensemble import RandomForestRegressor
            
            # Train model to get feature importance
            model = RandomForestRegressor(n_estimators=100, random_state=42)
            model.fit(X, y)
            
            # Get feature importance
            importance = pd.DataFrame({
                'feature': X.columns,
                'importance': model.feature_importances_
            }).sort_values('importance', ascending=False)
            
            # Select top features
            top_features = importance.head(20)['feature'].tolist()
            return X[top_features]
        
        elif method == 'recursive':
            from sklearn.feature_selection import RFECV
            
            selector = RFECV(
                estimator=RandomForestRegressor(n_estimators=50),
                step=1,
                cv=5,
                scoring='neg_mean_squared_error',
                n_jobs=-1
            )
            selector.fit(X, y)
            return X.loc[:, selector.support_]

Cross-Validation Strategy

class CrossValidation:
    def __init__(self):
        self.scores = {}
    
    def time_series_cv(self, X, y, n_splits=5):
        """Time series cross-validation for forecasting problems"""
        from sklearn.model_selection import TimeSeriesSplit
        
        tscv = TimeSeriesSplit(n_splits=n_splits)
        cv_scores = []
        
        for train_idx, val_idx in tscv.split(X):
            X_train, X_val = X.iloc[train_idx], X.iloc[val_idx]
            y_train, y_val = y.iloc[train_idx], y.iloc[val_idx]
            
            # Train and evaluate
            model = self.train_model(X_train, y_train)
            score = self.evaluate_model(model, X_val, y_val)
            cv_scores.append(score)
        
        return np.mean(cv_scores), np.std(cv_scores)
    
    def stratified_kfold(self, X, y, n_splits=5):
        """Stratified K-Fold for classification problems"""
        from sklearn.model_selection import StratifiedKFold
        
        skf = StratifiedKFold(n_splits=n_splits, shuffle=True, random_state=42)
        cv_scores = []
        
        for train_idx, val_idx in skf.split(X, y):
            X_train, X_val = X.iloc[train_idx], X.iloc[val_idx]
            y_train, y_val = y.iloc[train_idx], y.iloc[val_idx]
            
            model = self.train_model(X_train, y_train)
            score = self.evaluate_model(model, X_val, y_val)
            cv_scores.append(score)
        
        return cv_scores

Key Learnings & Insights

Critical Success Factors

1. Feature Engineering > Model Complexity

Investing time in creating meaningful features had 3x more impact on performance than hyperparameter tuning.

2. Ensemble Diversity Matters

Combining models with different biases (tree-based + linear + neural networks) improved robustness by 15%.

3. Validation Strategy is Crucial

Proper cross-validation prevented overfitting and provided reliable performance estimates.

4. Data Leakage Detection

Learning to identify and prevent data leakage was essential for trustworthy results.

Common Pitfalls Avoided

  • Target Leakage: Ensured no future information was used to predict past events
  • Overfitting Public LB: Used multiple validation strategies beyond public leaderboard
  • Ignoring Business Context: Considered practical implications of model errors
  • Feature Importance Misinterpretation: Used multiple methods to validate feature importance
