Dog Adoption Website

Responsive web platform connecting rescue dogs with loving families through modern web technologies

Spring 2025 HTML/CSS/JavaScript Fully Responsive

Project Overview

The Challenge

Animal shelters struggle with low visibility for adoptable dogs, complicated adoption processes, and inefficient communication between shelters and potential adopters.

Project Goal

Create an intuitive, mobile-friendly website that streamlines the dog adoption process while providing comprehensive information about available dogs and the adoption process.

Target Users

Potential pet adopters, animal shelters, rescue organizations, and volunteers looking to facilitate successful adoptions.

Design & User Experience

User-Centered Design Approach

1. User Research

Conducted interviews with 15 potential adopters and 3 shelter staff to understand pain points in the adoption process.

  • Key Finding: 80% of users wanted better filtering options
  • Insight: High-quality photos significantly increased inquiry rates
  • Need: Clear adoption process transparency

Interactive Filter System

// Advanced Filtering System
class DogFilter {
    constructor(dogs) {
        this.dogs = dogs;
        this.filters = {
            size: [],
            age: [],
            breed: [],
            gender: [],
            goodWith: [],
            energyLevel: [],
            specialNeeds: false
        };
    }
    
    applyFilters() {
        return this.dogs.filter(dog => {
            return Object.entries(this.filters).every(([key, value]) => {
                if (value.length === 0 && key !== 'specialNeeds') return true;
                
                switch(key) {
                    case 'size':
                        return value.includes(dog.size);
                    case 'age':
                        return this.checkAge(dog.age, value);
                    case 'breed':
                        return value.includes(dog.breed);
                    case 'gender':
                        return value.includes(dog.gender);
                    case 'goodWith':
                        return value.every(item => dog.goodWith.includes(item));
                    case 'energyLevel':
                        return value.includes(dog.energyLevel);
                    case 'specialNeeds':
                        return !value || dog.specialNeeds === value;
                    default:
                        return true;
                }
            });
        });
    }
    
    checkAge(dogAge, selectedAges) {
        const ageRanges = {
            'puppy': { min: 0, max: 1 },
            'young': { min: 1, max: 3 },
            'adult': { min: 3, max: 8 },
            'senior': { min: 8, max: Infinity }
        };
        
        return selectedAges.some(ageRange => {
            const range = ageRanges[ageRange];
            return dogAge >= range.min && dogAge < range.max;
        });
    }
    
    updateFilter(filterType, value) {
        if (filterType === 'specialNeeds') {
            this.filters[filterType] = value;
        } else {
            const index = this.filters[filterType].indexOf(value);
            if (index === -1) {
                this.filters[filterType].push(value);
            } else {
                this.filters[filterType].splice(index, 1);
            }
        }
        
        return this.applyFilters();
    }
    
    // Real-time search functionality
    searchByName(searchTerm) {
        return this.dogs.filter(dog => 
            dog.name.toLowerCase().includes(searchTerm.toLowerCase()) ||
            dog.breed.toLowerCase().includes(searchTerm.toLowerCase()) ||
            dog.description.toLowerCase().includes(searchTerm.toLowerCase())
        );
    }
}

Secure Contact Form

// Form Validation & Submission
class AdoptionForm {
    constructor() {
        this.form = document.getElementById('adoption-form');
        this.init();
    }
    
    init() {
        this.form.addEventListener('submit', this.handleSubmit.bind(this));
        this.setupValidation();
    }
    
    setupValidation() {
        const inputs = this.form.querySelectorAll('input, textarea, select');
        
        inputs.forEach(input => {
            input.addEventListener('blur', this.validateField.bind(this));
            input.addEventListener('input', this.clearError.bind(this));
        });
    }
    
    validateField(e) {
        const field = e.target;
        const value = field.value.trim();
        let isValid = true;
        let errorMessage = '';
        
        switch(field.name) {
            case 'email':
                isValid = this.validateEmail(value);
                errorMessage = 'Please enter a valid email address';
                break;
            case 'phone':
                isValid = this.validatePhone(value);
                errorMessage = 'Please enter a valid phone number';
                break;
            case 'zip':
                isValid = this.validateZip(value);
                errorMessage = 'Please enter a valid 5-digit ZIP code';
                break;
            case 'message':
                isValid = value.length >= 20;
                errorMessage = 'Please provide more details (minimum 20 characters)';
                break;
            default:
                if (field.required) {
                    isValid = value !== '';
                    errorMessage = 'This field is required';
                }
        }
        
        if (!isValid) {
            this.showError(field, errorMessage);
            return false;
        }
        
        this.clearError(field);
        return true;
    }
    
    validateEmail(email) {
        const re = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
        return re.test(email);
    }
    
    async handleSubmit(e) {
        e.preventDefault();
        
        // Validate all fields
        const fields = this.form.querySelectorAll('input, textarea, select');
        let allValid = true;
        
        for (const field of fields) {
            if (!this.validateField({ target: field })) {
                allValid = false;
            }
        }
        
        if (!allValid) {
            this.showFormError('Please fix the errors in the form');
            return;
        }
        
        // Prepare form data
        const formData = new FormData(this.form);
        const data = Object.fromEntries(formData);
        
        // Add anti-spam honeypot
        if (data.honeypot) {
            console.log('Spam detected');
            return;
        }
        
        // Show loading state
        this.setLoading(true);
        
        try {
            // Send to backend
            const response = await fetch('/api/adoption-inquiry', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify(data)
            });
            
            if (response.ok) {
                this.showSuccess();
                this.form.reset();
            } else {
                throw new Error('Submission failed');
            }
        } catch (error) {
            this.showFormError('There was an error submitting your inquiry. Please try again.');
        } finally {
            this.setLoading(false);
        }
    }
}

Accessibility Implementation

WCAG 2.1 AA Compliance

1. Semantic HTML

Proper use of HTML5 semantic elements for screen readers:

<header role="banner">
<nav role="navigation" aria-label="Main navigation">
<main role="main">
<article>
    <h1>Dog Profile: Max</h1>
    <img src="max.jpg" alt="Golden Retriever smiling at camera">
</article>
<aside role="complementary">
<footer role="contentinfo">

2. Keyboard Navigation

  • All interactive elements accessible via keyboard
  • Visible focus indicators for all focusable elements
  • Logical tab order following visual layout
  • Skip navigation links for screen reader users

3. Color & Contrast

  • Minimum contrast ratio of 4.5:1 for normal text
  • 3:1 contrast for large text and UI components
  • Color not used as sole means of conveying information
  • Dark mode support with system preferences

4. Screen Reader Optimizations

  • ARIA labels for complex widgets
  • Live regions for dynamic content updates
  • Proper heading hierarchy (h1-h6)
  • Descriptive link text and image alt text
